\subsection*{Installed Programs}
    the image comes with pre-installed programs (in addition to the used packages above).\\
    \begin{tabular}{m{7cm}m{7cm}}
    	\textbf{{\large Utility Programs}}
    	\begin{description}
    		\item [dolphin] small file browser
    		\item [urxvt] transparent terminal
    		\item [openssh] SSH connectivity
    		\item[scrot] screenshot script
    		\item[slock] simple lock screen script
    		\item[alsa-*] necessary ALSA packages
    		\item[latex-suite] required package for \TeX usage
    		\item[nitrogen] an application to set the background image from a selected folder
            \item[make\LaTeX] a tool to generate latex document templates
            \item[prmaker] a tool to initialize new software projects
    	\end{description}
    	&
    	\textbf{\large{Multimedia Application}}
    	\begin{description}
    		\item [mypaint] professional drawing tool
    		\item [flashplugin] required plugin to see certain elements in web browser
    		\item [audacity] audio manipulating program \textit{(ardour in validation\ldots)}
    		\item [gimp] graphic and image manipulating program
            \item [inkscape] a drawing tool for SVG graphics
    		\item [vlc] cross platform video player
    		\item [moc] music player on console
    		\item [feh] Image Viewer
    	\end{description}
    	\\
    	\textbf{{\large Office Application}}
    	\begin{description}
    		\item[epiphany] simple web browser \textit{(others in validation\ldots)}
    		\item [evince] PDF Viewer
    		\item [transmission-qt] torrent client
    		\item [vim] advanced text editor in console
    		\item [TexStudio] \TeX Editor
       		\item [octave] tool for numerical computations
            \item[R] the stochastic tool and language R
               
    	\end{description}
    	&
    	\textbf{{\large Development Application}}
    	\begin{description}
    		\item[gcc] c/c++ compiler
    		\item[base-devel] base package for make etc
    		\item[git] version control system
    		\item[jdk8-openjdk] java development kit
    		\item[umlet] UML editor \textit{(others in validation\ldots)}
            \item[cmake] a cross platform make system
            \item[gprof] a profiling program
            \item[gdb] a debug interface
    	\end{description}
    	\\
    
    	\textbf{{\large Communication Application}}
    	\begin{description}
    		\item [irssi] irc client on console
    		\item [skype] multi-user internet chat program
    	\end{description}
    \end{tabular}
